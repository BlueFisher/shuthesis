\begin{denotation}[2.5cm]
\item[$\mathcal{T}$] 张量
\item[$H$] 超图
\item[$\mathcal{A}(H)$] 超图 $H$ 的邻接张量
\item[$\mathcal{L}(H)$] 超图 $H$ 的拉普拉斯张量
\item[$\mathcal{Q}(H)$] 超图 $H$ 的无符号拉普拉斯张量
\item[$\rho$] 谱半径
\item[$G$] 图
\item[$\kappa$] 连通度
\item[$\chi$] 染色数
\item[$\Delta$] 最大度
\item[$\delta$] 最小度
\end{denotation}
